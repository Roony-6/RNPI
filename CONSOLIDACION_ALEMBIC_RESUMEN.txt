💻 COMANDOS PARA FUTURAS MIGRACIONES
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Después de cambiar modelos en app/models/core.py o app/models/catalogos.py:

```bash
# 1. Generar migración (auto-detecta cambios)
export DATABASE_URL="postgresql://postgres@localhost/rnpi"
alembic revision --autogenerate -m "Descripción breve"

# 2. Revisar archivo generado en alembic/versions/
# 3. Aplicar a BD
alembic upgrade head

# Verificar estado
alembic current
alembic history --verbose
```
